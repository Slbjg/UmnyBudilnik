package com.example.lab1

import android.app.Service
import android.content.Context
import android.content.Intent
import android.hardware.Sensor
import android.hardware.SensorEvent
import android.hardware.SensorEventListener
import android.hardware.SensorManager
import android.media.MediaPlayer
import android.media.RingtoneManager
import android.os.Build
import android.os.IBinder
import android.os.VibrationEffect
import android.os.Vibrator
import android.util.Log

class AlarmService : Service(), SensorEventListener {

    private lateinit var sensorManager: SensorManager
    private lateinit var vibrator: Vibrator
    private var accelerometer: Sensor? = null
    private var mediaPlayer: MediaPlayer? = null
    
    private var shakeCount = 0
    private val REQUIRED_SHAKES = 3
    private val SHAKE_THRESHOLD = 100f
    private val SHAKE_COOLDOWN = 800L
    private val STABILIZATION_DELAY_MS = 2000L
    
    private var lastShakeTime = 0L
    private var lastUpdateTime = 0L
    private var lastX = 0f
    private var lastY = 0f
    private var lastZ = 0f
    private var alarmStartTime = 0L
    private var isAlarmActive = false

    override fun onCreate() {
        super.onCreate()
        sensorManager = getSystemService(Context.SENSOR_SERVICE) as SensorManager
        vibrator = getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
        accelerometer = sensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER)
        Log.d("SHAKE", "Сервис создан")
    }

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        if (intent?.action == "STOP_ALARM") {
            stopAlarm()
            return START_NOT_STICKY
        }

        if (isAlarmActive) return START_NOT_STICKY

        isAlarmActive = true
        shakeCount = 0
        alarmStartTime = System.currentTimeMillis()
        lastUpdateTime = 0L
        lastShakeTime = 0L
        lastX = 0f
        lastY = 0f
        lastZ = 0f

        sendBroadcast(Intent("ALARM_STARTED"))

        val ringtoneUri = RingtoneManager.getDefaultUri(RingtoneManager.TYPE_ALARM)
        mediaPlayer = MediaPlayer.create(this, ringtoneUri).apply {
            isLooping = true
            start()
        }

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            vibrator.vibrate(VibrationEffect.createWaveform(longArrayOf(0, 500, 500), 0))
        } else {
            @Suppress("DEPRECATION")
            vibrator.vibrate(longArrayOf(0, 500, 500), 0)
        }

        accelerometer?.also {
            sensorManager.registerListener(this, it, SensorManager.SENSOR_DELAY_GAME)
        }

        return START_STICKY
    }

    override fun onSensorChanged(event: SensorEvent?) {
        if (!isAlarmActive) return
        if (event?.sensor?.type != Sensor.TYPE_ACCELEROMETER) return

        val currentTime = System.currentTimeMillis()
        if (currentTime - alarmStartTime < STABILIZATION_DELAY_MS) return

        val x = event.values[0]
        val y = event.values[1]
        val z = event.values[2]

        if (lastUpdateTime == 0L) {
            lastUpdateTime = currentTime
            lastX = x
            lastY = y
            lastZ = z
            return
        }

        if (currentTime - lastUpdateTime >= 100) {
            val diffTime = currentTime - lastUpdateTime
            lastUpdateTime = currentTime

            val deltaX = Math.abs(x - lastX)
            val deltaY = Math.abs(y - lastY)
            val deltaZ = Math.abs(z - lastZ)
            val maxDelta = maxOf(deltaX, deltaY, deltaZ)
            val speed = maxDelta / diffTime * 10000

            if (speed > SHAKE_THRESHOLD && currentTime - lastShakeTime > SHAKE_COOLDOWN) {
                shakeCount++
                lastShakeTime = currentTime

                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                    vibrator.vibrate(VibrationEffect.createOneShot(100, VibrationEffect.DEFAULT_AMPLITUDE))
                } else {
                    @Suppress("DEPRECATION")
                    vibrator.vibrate(100)
                }

                val intent = Intent("SHAKE_COUNT_UPDATE")
                intent.putExtra("count", shakeCount)
                sendBroadcast(intent)

                if (shakeCount >= REQUIRED_SHAKES) {
                    stopAlarm()
                }
            }

            lastX = x
            lastY = y
            lastZ = z
        }
    }

    private fun stopAlarm() {
        if (!isAlarmActive) return
        
        val intent = Intent("ALARM_STOPPED")
        intent.putExtra("count", shakeCount)
        sendBroadcast(intent)
        
        mediaPlayer?.stop()
        mediaPlayer?.release()
        mediaPlayer = null
        vibrator.cancel()
        sensorManager.unregisterListener(this)
        isAlarmActive = false
        stopSelf()
    }

    override fun onAccuracyChanged(sensor: Sensor?, accuracy: Int) {}

    override fun onDestroy() {
        super.onDestroy()
        mediaPlayer?.release()
        vibrator.cancel()
        try { sensorManager.unregisterListener(this) } catch (e: Exception) {}
    }

    override fun onBind(intent: Intent?): IBinder? = null
}

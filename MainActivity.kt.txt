package com.example.lab1

import android.app.AlarmManager
import android.app.PendingIntent
import android.content.Context
import android.content.Intent
import android.os.Build
import android.os.Bundle
import android.provider.Settings
import android.widget.Button
import android.widget.TextView
import android.widget.TimePicker
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import java.util.Calendar

class MainActivity : AppCompatActivity() {

    private lateinit var timePicker: TimePicker
    private lateinit var setAlarmButton: Button
    private lateinit var statusText: TextView
    private lateinit var shakeCountText: TextView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        timePicker = findViewById(R.id.timePicker)
        setAlarmButton = findViewById(R.id.setAlarmButton)
        statusText = findViewById(R.id.statusText)
        shakeCountText = findViewById(R.id.shakeCountText)

        setAlarmButton.setOnClickListener {
            setAlarm()
        }

        checkExactAlarmPermission()
    }

    private fun setAlarm() {
        val calendar = Calendar.getInstance().apply {
            set(Calendar.HOUR_OF_DAY, timePicker.hour)
            set(Calendar.MINUTE, timePicker.minute)
            set(Calendar.SECOND, 0)

            if (timeInMillis <= System.currentTimeMillis()) {
                add(Calendar.DAY_OF_YEAR, 1)
            }
        }

        val intent = Intent(this, AlarmReceiver::class.java)
        val pendingIntent = PendingIntent.getBroadcast(
            this, 0, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )

        val alarmManager = getSystemService(Context.ALARM_SERVICE) as AlarmManager

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            if (alarmManager.canScheduleExactAlarms()) {
                alarmManager.setExactAndAllowWhileIdle(
                    AlarmManager.RTC_WAKEUP,
                    calendar.timeInMillis,
                    pendingIntent
                )
            } else {
                Toast.makeText(this, "Нет разрешения", Toast.LENGTH_LONG).show()
                return
            }
        } else {
            alarmManager.setExactAndAllowWhileIdle(
                AlarmManager.RTC_WAKEUP,
                calendar.timeInMillis,
                pendingIntent
            )
        }

        val timeString = String.format("%02d:%02d", timePicker.hour, timePicker.minute)
        statusText.text = "Будильник заведён на $timeString"
        Toast.makeText(this, "Будильник на $timeString", Toast.LENGTH_SHORT).show()
    }

    private fun checkExactAlarmPermission() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            val alarmManager = getSystemService(Context.ALARM_SERVICE) as AlarmManager
            if (!alarmManager.canScheduleExactAlarms()) {
                startActivity(Intent(Settings.ACTION_REQUEST_SCHEDULE_EXACT_ALARM))
            }
        }
    }

    private val shakeReceiver = object : android.content.BroadcastReceiver() {
        override fun onReceive(context: Context, intent: Intent) {
            when (intent.action) {
                "SHAKE_COUNT_UPDATE" -> {
                    val count = intent.getIntExtra("count", 0)
                    shakeCountText.text = "[$count/3]"
                }
                "ALARM_STARTED" -> {
                    statusText.text = "БУДИЛЬНИК ЗВОНИТ! Встряхните 3 раза!"
                    shakeCountText.text = "[0/3]"
                }
                "ALARM_STOPPED" -> {
                    val count = intent.getIntExtra("count", 0)
                    statusText.text = "Будильник выключен! ($count встряхиваний)"
                    shakeCountText.text = "[$count/3]"
                }
            }
        }
    }

    override fun onResume() {
        super.onResume()
        android.content.IntentFilter().apply {
            addAction("SHAKE_COUNT_UPDATE")
            addAction("ALARM_STARTED")
            addAction("ALARM_STOPPED")
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                registerReceiver(shakeReceiver, this, Context.RECEIVER_NOT_EXPORTED)
            } else {
                @Suppress("UnspecifiedRegisterReceiverFlag")
                registerReceiver(shakeReceiver, this)
            }
        }
    }

    override fun onPause() {
        super.onPause()
        try {
            unregisterReceiver(shakeReceiver)
        } catch (e: Exception) {
        }
    }
}
